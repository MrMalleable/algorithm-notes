    2399 检查相同字母间的距离

    这个题目比较简单,主要是使用一个数组来记录上一次出现的位置，然后遍历到第二个出现的地方之后，看下两次出现位置之间的距离是否等于distance数组相应索引处的值。

    代码如下：

    class Solution {
    public boolean checkDistances(String s, int[] distance) {
        // 存放26个字母第一次出现的位置
        int[] cnt = new int[26];
        for(int i = 0; i < 26; i++){
            cnt[i] = -1;
        }
        char[] arr = s.toCharArray();
        for(int i = 0; i < arr.length; i++){
            char c = arr[i];
            // 第一次出现记录下位置
            if(cnt[c-'a']==-1){
                cnt[c-'a'] = i;
            }else{
                // 计算和第一次出现位置的距离
                int dis = i - cnt[c-'a'] - 1;
                if(dis != distance[c-'a']){
                    // 如果不想等直接返回false
                    return false;
                }
            }
        }
        return true;
    }
}

    2400 恰好移动k步到达某一位置的方法数目

    这道题目我使用暴力dfs真的沙雕，对于小数据还能过，但是一道那种奇怪的用例就直接超时，当时做的时候虽然知道这样不是一个好办法，但就是想不出来有啥好办法，导致心态崩溃。

    这里好好贴下TsReaper大佬的解法：

    数学： 称startPos指向endPos的方向为正方向，startPos远离endPos的方向为负方向。设从startPos出发，往正方向走了a步，往负方向走了（k-a）步后到达endPos，根据组合书的定义可知答案为C(k,a)(k步里面选a步走正方向)。

    记d = abs(endPos - startPos) , 有方程 a - (k - a) = d, 得到a = (d+k) / 2。因此首先判断是否（d+k）是否为偶数（这样才能求出整数a),以及d <= k(否则走不到)，然后求组合数即可。

    代码如下：

    class Solution{
    const int MOD = 1000000007;
public:
    int numberOfWays(int startPos, int endPos, int k){
        int d = abs(startPos - endPos);
        if((d+k) % 2 == 1 || d > k){
            return 0;
        }

        vector<vector<long long>> f;
        f.resize(k+1, vector<long long>(k+1));

        for(int i = 0; i <= k ; i++){
            f[i][0] = 1;
            for(int j = 1; j <= i; j++){
                f[i][j] = (f[i-1][j] + f[i-1][j-1]) % MOD;
            }
        }
        return f[k][(d+k)/ 2];
    }
};

    最长优雅子数组

    这道题目其实是一道很经典的滑动窗口的题目，题目的意思是需要我们找到一个连续的子数组，这些子数组每个数转化成二进制时各个位上的1都不会重复，所以我们用一个滑动窗口来维护，如果右移一个元素导致哪一个二进制位出现第二次1，那么左边的指针就要不停地移动，直至不出现重复的情况。

    代码如下：

    class Solution {
    // 暴力竟然能过！
    public int longestNiceSubarray(int[] nums) {
        int ret = 1;
        for(int i = 0; i < nums.length; i++){
            for(int j = i + 1; j < nums.length; j++){
                if(elegant(nums, i, j)){
                    ret = Math.max(ret, j - i + 1);
                }else{
                    break;
                }
            }
        }
        return ret;
    }

    // 判断startIndex到endIndex是不是一个优雅数组
    private boolean elegant(int[] nums, int startIndex, int endIndex){
        for(int i = startIndex; i < endIndex; i++){
            if((nums[endIndex] & nums[i]) != 0){
                return false;
            }
        }
        return true;
    }
}

    2402 会议室III

    抄下大佬的思路：模拟=最小堆+最小堆

    class Node{
    long endTime;
    int roomID;

    Node(){}
    Node(long endTime, int roomID){
        this.endTime = endTime;
        this.roomID = rootID;
    }
}

class Solution{
    public int mostBooked(int n , int[][] meetings){
        int[] room_cnt = new int[n];
        Arrays.sort(meetings, (a,b)->{
            if(a[0] != b[0]){
                return a[0] - b[0];
            }
            return a[1] - b[1];
        });
        PriorityQueue<Integer> min_room = new PriorityQueue<>(new Comparator<>(){
            public int compare(Integer a, Integer b)
            {
                return a - b;
            }
        });
        PriorityQueue<Node> min_meet = new PriorityQueue<>(new Comparator<>(){
            public int compare(Node a, Node b)
            {
                if (a.endTime != b.endTime){
                    return (int)(a.endTime - b.endTime);
                }
                return a.roomID - b.roomID;
            }
        });

        for (int ID = 0; ID < n; ID ++)
        {
            min_room.offer(ID);
        }
        for (int [] me: meetings)
        {
            int s = me[0], e = me[1];
            //---- 已经开完会的会议室，空闲出来
            while (!min_meet.isEmpty() && min_meet.peek().endTime <= s)
            {
                Node tmp = min_meet.poll();
                long endTime = tmp.endTime;
                int roomID = tmp.roomID;
                min_room.offer(roomID);
            }
            //---- 如果有空闲的会议室
            if (!min_room.isEmpty())
            {
                int roomID = min_room.poll();
                room_cnt[roomID] ++;
                min_meet.offer(new Node(e, roomID));
            }
            //---- 如果没有会议室
            else
            {
                Node tmp = min_meet.poll();
                long endTime = tmp.endTime;
                int roomID = tmp.roomID;
                room_cnt[roomID] ++;
                min_meet.offer(new Node(endTime + e - s, roomID));
            }

        }

        int max_val = 0;
        int max_i = 0;
        for (int i = 0; i < n; i ++)
        {
            if (room_cnt[i] > max_val)
            {
                max_val = room_cnt[i];
                max_i = i;
            }
        }
        return max_i;
    }
}
